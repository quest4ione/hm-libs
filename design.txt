# General structure
type: string  // object type
_id: string  // format: type_id
audit: [
  {
    time: int // time in seconds since epoch at creation
    operation: object // the update/insert query
    script: string // the script that edited the db
    user: string // the user that initiated the db edit / called the script
    memo: string // a memo describing by what / why stuff was changed 
  }
  ...
]
...object-specific values

# Types
## User
type: string // "user"
_id: string // "user_{name}"
audit: []

loc: string || null // the users loc if we know that
// TODO: maybe some kind of script run / upgrade / gc balance log for tracking >:)

## Script
type: string // "script"
_id: string // "script_{owner}_{name}"
audit: []

call_log: [ // maybe delete old logs (spam detection / after 30 days)
  {
    time: int // time in seconds since epoch at creation
    run_id: string // the _RUN_ID
    success: bool || null // wether the script ran successfully. null means unknown
    context: { // the context object
      bot_brain: bool // wether run from a bot brain
      caller: string // the caller
      calling_script: string || null // the calling script, if any
      this_script: string // this script,
      cols: int // the calling users terminal cols
      rows: int // the calling users terminal rows
    }
    args: object // input args
    result: object || array || string || bool || null // the return object
    stack: [ // array of known stack (incomplete) contexts. order is outer to inner. subarrays indicate gaps of unknown length
      [
        object // partial context object
        ...
      ]
      ...
    ]
    data: object // any additional data provided useful for debugging
  }
  ...
]
